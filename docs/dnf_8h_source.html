25class DelayLine {
26public:
27 void init(size_t delaySamples)
28 {
29 delaySamples_ = delaySamples;
30 buffer_ = std::deque<float>(delaySamples_, 0.0f);
31 }
32
33 float process(float input) {
34 // Output is the oldest value (front of deque)
35 float output = buffer_.front();
36 buffer_.pop_front();
37
38 // Push new input to the back
39 buffer_.push_back(input);
40
41 return output;
42 }
43
44 float get(int i) const {
45 return buffer_[i];
46 }
47
48 float getNewest() const {
49 return buffer_.back();
50 }
51
52private:
53 size_t delaySamples_ = 0;
54 std::deque<float> buffer_;
55};

74 Net(int nLayers, int nInput, bool withBias = false) {
75 // calc an exp reduction of the numbers always reaching 1
76 const float b = (float)exp(log(nInput)/(nLayers-1));
77 int inputNeurons = nInput;
78 for(int i=1;i<nLayers;i++) {
79 int outputNeurons = (int)ceil(nInput / pow(b,i));
80 if (i == (nLayers-1)) outputNeurons = 1;
81 char tmp[256];
82 sprintf(tmp,"fc%d_%d_%d",i,inputNeurons,outputNeurons);
83 if (debugOutput)
84 fprintf(stderr,"Creating FC layer: %s\n",tmp);
85 torch::nn::Linear ll = register_module(
86 tmp,
87 torch::nn::Linear(torch::nn::LinearOptions(inputNeurons, outputNeurons).bias(withBias))
88 );
89 torch::nn::init::xavier_uniform_(ll->weight,xavierGain);
90 if (withBias) torch::nn::init::constant_(ll->bias, 0.0);
91 fc.push_back(ll);
92 if (1 == outputNeurons) break;
93 inputNeurons = outputNeurons;
94 }
95 }

97 torch::Tensor forward(torch::Tensor x, ActMethod am) {
98 for(auto& f:fc) {
99 switch (am) {
100 default:
101 case Act_Tanh:
102 x = torch::atan(f->forward(x));
103 break;
104 case Act_Sigmoid:
105 x = torch::sigmoid(f->forward(x));
106 break;
107 case Act_ReLU:
108 x = torch::relu(f->forward(x));
109 break;
110 case Act_NONE:
111 x = f->forward(x);
112 break;
113 }
114 }
115 return x;
116 }
